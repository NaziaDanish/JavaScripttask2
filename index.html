Q2- Write a program that takes two numbers and displays the result of all basic arithmetic operations.

// Pehle 2 number lenge input lene k liye.

var num1 = prompt("Pehla number daalainge:");

var num2 = prompt("Dosra number daalainge:");

// Numbers ko integer mein convert karna

num1 = parseInt(num1);

num2 = parseInt(num2);

// Operations

var addition = num1 + num2;

var subtraction = num1 - num2;

var multiplication = num1 * num2;

var division = num1 / num2;

// Results in Alert

alert("Addition: " + addition);

alert("Subtraction: " + subtraction);

alert("Multiplication: " + multiplication);

alert("Division: " + division);

// Console.log

console.log("Addition: " + addition);
